%% ================= WALLET =================
subgraph W["WALLET — build & prove"]
  direction TB
  WK{spend type}
  WK -->|transparent| WA["Pedersen commit + Schnorr ownership/value"]
  WK -->|confidential ZK| WB["STARK prove: tree-membership + range + balance (hidden amounts)"]
  WK -->|unlinkable| WC["recipient-secret note: pk=H(nk), nf=H(nk,rho)"]
  WK -->|sender-anon ring| WD["Triptych one-out-of-many + key-image"]
  WK -->|vault| WE["lock deposit (private staking)"]
  WK -->|cross-chain| WF["BTC HTLC / Nano scriptless adaptor"]
  WK -->|post-quantum| WG["ML-KEM-768 stealth + lattice commitment"]
  WA & WB & WC & WD & WE & WF & WG --> WTX["signed tx + proof, stealth one-time output"]
end

WTX -->|"POST /submittx"| RPC

%% ================= NODE / MEMPOOL =================
subgraph N["NODE — admission"]
  direction TB
  RPC["RPC"] --> ADM["mempool admit (off-lock, parallel)"]
  ADM --> CHK["verify proofs · value conservation · double-spend (shared nullifier set)"]
  CHK -->|"fee floor · bounded 5000 · fee-priority evict"| POOL[("mempool")]
end

POOL -->|"Dandelion++ stem → fluff"| GOSS

%% ================= P2P MESH =================
subgraph P["P2P MESH — self-discovering, censorship-resistant"]
  direction LR
  GOSS["gossip blocks + txs"] --- DISC["PEX + self-address discovery (addr-me, 2+ votes)"]
  DISC --- BOOK[("persistent address book")]
  GOSS --- TOR["Tor / .onion (NAT traversal)"]
end

POOL -->|"select up to 1000 txs"| SEL

%% ================= MINER =================
subgraph M["MINER — must be a FULL NODE"]
  direction TB
  SEL["select txs"] --> TMPL["build block template"]
  TMPL --> POR["Proof-of-Retrievability: answer k random body challenges (read history)"]
  POR -.->|"pruned node → no body → CANNOT mine"| STOP(["rejected: not full node"])
  POR --> ROOTS["compute header roots"]
  ROOTS --> POW["RandomX PoW — paced to block time"]
  POW --> BLK["mined block"]
end

BLK --> VAL

%% ================= VALIDATION =================
subgraph V["BLOCK VALIDATION — every node, even pruned"]
  direction TB
  V1["PoW + difficulty (LWMA) + median-time"] --> V2["PoR proofs vs stored headers"]
  V2 --> V3["merkle root + NumTxs"]
  V3 --> V4["per-tx proofs: STARK / ring / Pedersen / PQ (parallel)"]
  V4 --> V5["conservation + nullifier no-double-spend"]
  V5 --> V6["accumulator + Null + CM + PQ checkpoints match header"]
end

VAL --> APPLY["APPLY block (atomic, reorg-safe snapshot)"]

%% ================= STATE =================
subgraph S["STATE — constant-size, committed"]
  direction LR
  ACC[("RSA-2048 accumulator (all coins, O(1))")]
  NUL[("nullifier set (spent)")]
  CM[("epoch-sharded Poseidon commitment tree")]
  PQA[("PQ anonymity accumulator")]
  EXT[("vaults · swap contracts · incentive pool")]
end

APPLY --> ACC & NUL & CM & PQA & EXT

%% ================= HEADER + CHAIN =================
ACC & NUL & CM & PQA --> HDR["HEADER COMMITMENTS: AccRoot · NullRoot · CMRoot · PQAccRoot · PoRRoot · NumTxs"]
HDR -.->|"bound by PoW (in preimage)"| ROOTS
APPLY --> CHAIN[("canonical chain")]

%% gossip the new block out + convergence
BLK -->|"broadcast block"| GOSS
GOSS -->|"peers validate + apply"| VAL
CHAIN -->|"fork-choice = most cumulative work → reorg + replay"| CHAIN

%% storage / new-node onboarding
CHAIN --> PRUNE["3-tier pruning + state snapshot"]
PRUNE -.->|"snapshot sync serves new nodes"| GOSS

%% spend anchors back into the tree
CM -.->|"recent CMRoot = spend anchor"| WB
